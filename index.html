package com.eventsystem.security; import io.jsonwebtoken.Jwts; import io.jsonwebtoken.security.Keys; import javax.crypto.SecretKey; public class JwtUtil { private static final SecretKey secretKey = Keys.hmacShaKeyFor( "mysecretkeymysecretkeymysecretkey12345" .getBytes() ); public static String generateToken( String email){ return Jwts.builder() .subject(email) .signWith(secretKey) .compact(); } public static String getEmail( String token){ return Jwts.parser() .verifyWith(secretKey) .build() .parseSignedClaims(token) .getPayload() .getSubject(); } }
